### Usage
```js static
import { Label } from '@codespec/react-toolset'
```

### Basic Usage
```js
<Label>My Label</Label>
```

### Required Field Label
```js
<Label required={true}>
  Required Label:
</Label>
```

### Label For
```js
<div>
  <Label htmlFor="simple-button">
    Label For:
    <Button id="simple-button">Simple Button</Button>
  </Label>
</div>
```
