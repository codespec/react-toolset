### Usage

```jsx static
import { Dropdown } from '@codespec/react-toolset'
```

### Basic Dropdown

```jsx
<div>
  <div>
    <Dropdown>
      Dropdown Label
      <option>Option 1</option>
      <option>Option 2</option>
    </Dropdown>
  </div>
  <br />
  <div>
    <Dropdown>
      Dropdown with Divider
      <option>Section 1</option>
      <divider />
      <option>Section 2</option>
    </Dropdown>
  </div>
</div>
```
