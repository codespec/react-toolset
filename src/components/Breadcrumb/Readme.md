### Usage

```jsx static
import { Breadcrumb } from '@codespec/react-toolset'
```

### Basic Breadcrumb

```jsx
<Breadcrumb>
  <a href="/">Home</a>
  <a href="/route">Route</a>
  You are here
</Breadcrumb>
```

### Custom Breadcrumb

```jsx
<Breadcrumb className='Breadcrumb--double'>
  <a href="/">Home</a>
  <a href="/route">Route</a>
  You are here
</Breadcrumb>
```
